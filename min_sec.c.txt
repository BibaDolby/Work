#include <stdio.h>

#define SEC_PER_MIN 60


int main()

{

    int sec, min, left;

    printf("Перевод секунд в мин и сек\n");

    printf("Введите количество секунд (<=0 для выхода):\n");

    scanf("%d", &sec);

    while (sec > 0)

    {

        min = sec / SEC_PER_MIN;

        left = sec % SEC_PER_MIN;

        printf("%d sec - это %d min и %d sec\n", sec, min, left);

        printf("Введите следующее значение (<=0 для выхода): ");

        scanf("%d", &sec);

    }

    printf("Готово!\n");

    return 0;

}
