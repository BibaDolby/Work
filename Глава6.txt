int main() 
{
    char c = 'A';
    int i, j;
    for(i = 0; i < 6; ++i)
    {
        for(j = 0; j < i + 1; ++j)
        {
            printf("%c", c);
            ++c;
        }
        printf("%c", '\n');
    }
}
